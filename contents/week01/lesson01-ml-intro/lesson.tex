\begin{definition}{机器学习}{w01l01-machine-learning}
机器学习研究的是如何利用数据来学习一个参数化函数，使其在给定输入时输出合理预测。
\end{definition}

一个典型的经验风险最小化目标可以写成
\begin{equation}
  \theta^\star = \argmin_{\theta \in \R^d}
  \frac{1}{N}\sum_{i=1}^{N}\ell\paren{f_{\theta}(x_i), y_i}.
  \label{eq:w01l01-empirical-risk}
\end{equation}

\begin{claim}{训练目标}{w01l01-optimization}
只要模型、损失函数和优化器都明确，后续课程的大部分整理都可以复用同一套写作骨架。
\end{claim}

\begin{algorithm}[htbp]
  \caption{梯度下降骨架}
  \label{alg:w01l01-gradient-descent}
  \KwInput{训练数据 $\mathcal{D}$，学习率 $\eta$，初始参数 $\theta^{(0)}$}
  \KwOutput{训练后的参数 $\theta^\star$}
  \For{$t \leftarrow 0$ \KwTo $T-1$}{
    计算梯度 $g^{(t)} \leftarrow \grad_{\theta} L\paren{\theta^{(t)}}$\;
    更新参数 $\theta^{(t+1)} \leftarrow \theta^{(t)} - \eta g^{(t)}$\;
  }
  \KwRet{$\theta^{(T)}$}
\end{algorithm}

\begin{lstlisting}[style=python,caption={训练循环代码占位},label={lst:w01l01-training-loop}]
def train_one_epoch(model, dataloader, optimizer):
    model.train()
    for batch in dataloader:
        loss = model.compute_loss(batch)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
\end{lstlisting}
